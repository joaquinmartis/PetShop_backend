-- virtualpet.init.sql
-- Archivo de inicialización SQL Estándar.
-- Contiene la estructura de la base de datos 'virtualpet' (Esquemas, Tablas, Claves, Índices y Funciones).

-----------------------------------
-- 1. ESQUEMAS (Schemas)
-----------------------------------

-- Crear esquemas (si la BDD lo soporta, si no, las tablas se crean directamente en el esquema principal)
-- En SQL estándar, a veces se usan como prefijos de tabla: 'esquema.tabla'.

CREATE SCHEMA IF NOT EXISTS user_management;
CREATE SCHEMA IF NOT EXISTS product_catalog;
CREATE SCHEMA IF NOT EXISTS cart;
CREATE SCHEMA IF NOT EXISTS order_management;
CREATE SCHEMA IF NOT EXISTS notification_management;

---

-----------------------------------
-- 2. FUNCIONES / PROCEDIMIENTOS (Simplificados/Adaptados)
-----------------------------------

-- En SQL estándar no hay un 'UPDATE_UPDATED_AT' trigger tan universal como en Postgres.
-- Se define una función genérica que debería ser adaptada al lenguaje procedural específico
-- de la BDD de destino (ej: PL/SQL para Oracle, T-SQL para SQL Server, etc.).
-- **NOTA:** Estas definiciones son *declarativas* y requieren ser implementadas o adaptadas.

-- Función para actualizar el campo 'updated_at' (Implementación variará por BDD)
-- CREATE FUNCTION update_updated_at_column()
RETURNS TIMESTAMP
BEGIN
    RETURN CURRENT_TIMESTAMP;
END;

-- Función específica para actualizar el timestamp del carrito (Implementación variará por BDD)
CREATE FUNCTION update_cart_timestamp(cart_id_param BIGINT)
RETURNS VOID
BEGIN
    UPDATE cart.carts
    SET updated_at = CURRENT_TIMESTAMP
    WHERE id = cart_id_param;
END;

-----------------------------------
-- 3. TABLAS Y SECUENCIAS (Tablas, Claves Primarias y Restricciones)
-----------------------------------

-- user_management
CREATE TABLE user_management.roles (
    id INT NOT NULL,
    name VARCHAR(50) NOT NULL,
    description TEXT,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT roles_pkey PRIMARY KEY (id),
    CONSTRAINT roles_name_key UNIQUE (name)
);

CREATE SEQUENCE user_management.roles_id_seq START WITH 1 INCREMENT BY 1;
-- ALTER TABLE user_management.roles ALTER COLUMN id SET DEFAULT NEXT VALUE FOR user_management.roles_id_seq; -- Sintaxis adaptable


CREATE TABLE user_management.users (
    id INT NOT NULL,
    email VARCHAR(100) NOT NULL,
    password_hash TEXT NOT NULL,
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    address TEXT NOT NULL,
    role_id INT NOT NULL,
    is_active BOOLEAN DEFAULT TRUE NOT NULL,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT users_pkey PRIMARY KEY (id),
    CONSTRAINT users_email_key UNIQUE (email),
    CONSTRAINT users_role_id_fkey FOREIGN KEY (role_id) REFERENCES user_management.roles(id)
);

CREATE SEQUENCE user_management.users_id_seq START WITH 1 INCREMENT BY 1;
-- ALTER TABLE user_management.users ALTER COLUMN id SET DEFAULT NEXT VALUE FOR user_management.users_id_seq;


-- product_catalog
CREATE TABLE product_catalog.categories (
    id INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    active BOOLEAN DEFAULT TRUE NOT NULL,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT categories_pkey PRIMARY KEY (id),
    CONSTRAINT categories_name_key UNIQUE (name)
);

CREATE SEQUENCE product_catalog.categories_id_seq START WITH 1 INCREMENT BY 1;


CREATE TABLE product_catalog.products (
    id INT NOT NULL,
    name VARCHAR(150) NOT NULL,
    description TEXT,
    price NUMERIC(10,2) NOT NULL,
    stock INT DEFAULT 0 NOT NULL,
    category_id INT NOT NULL,
    image_url VARCHAR(255),
    active BOOLEAN DEFAULT TRUE NOT NULL,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT products_pkey PRIMARY KEY (id),
    CONSTRAINT products_price_check CHECK (price >= 0),
    CONSTRAINT products_stock_check CHECK (stock >= 0),
    CONSTRAINT products_category_id_fkey FOREIGN KEY (category_id) REFERENCES product_catalog.categories(id)
);

CREATE SEQUENCE product_catalog.products_id_seq START WITH 1 INCREMENT BY 1;


-- cart
CREATE TABLE cart.carts (
    id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT carts_pkey PRIMARY KEY (id),
    CONSTRAINT unique_user_cart UNIQUE (user_id)
    -- FOREIGN KEY a user_management.users se omite por la arquitectura modular
);

CREATE SEQUENCE cart.carts_id_seq START WITH 1 INCREMENT BY 1;


CREATE TABLE cart.cart_items (
    id BIGINT NOT NULL,
    cart_id BIGINT NOT NULL,
    product_id BIGINT NOT NULL,
    quantity INT NOT NULL,
    unit_price_snapshot NUMERIC(10,2) NOT NULL,
    product_name_snapshot VARCHAR(150) NOT NULL,
    added_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT cart_items_pkey PRIMARY KEY (id),
    CONSTRAINT unique_cart_product UNIQUE (cart_id, product_id),
    CONSTRAINT cart_items_quantity_check CHECK (quantity > 0),
    CONSTRAINT cart_items_unit_price_snapshot_check CHECK (unit_price_snapshot >= 0),
    CONSTRAINT cart_items_cart_id_fkey FOREIGN KEY (cart_id) REFERENCES cart.carts(id) ON DELETE CASCADE
    -- FOREIGN KEY a product_catalog.products se omite por la arquitectura modular
);

CREATE SEQUENCE cart.cart_items_id_seq START WITH 1 INCREMENT BY 1;


-- order_management
CREATE TABLE order_management.orders (
    id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    status VARCHAR(50) NOT NULL,
    total NUMERIC(10,2) NOT NULL,
    shipping_method VARCHAR(20),
    shipping_id BIGINT,
    shipping_address TEXT NOT NULL,
    customer_name VARCHAR(200) NOT NULL,
    customer_email VARCHAR(100),
    customer_phone VARCHAR(20),
    notes TEXT,
    cancellation_reason VARCHAR(200),
    cancelled_at TIMESTAMP WITHOUT TIME ZONE,
    cancelled_by VARCHAR(20),
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT orders_pkey PRIMARY KEY (id),
    CONSTRAINT orders_total_check CHECK (total >= 0),
    CONSTRAINT check_status_values CHECK (status IN ('PENDING_VALIDATION', 'CONFIRMED', 'READY_TO_SHIP', 'SHIPPED', 'DELIVERED', 'CANCELLED')),
    CONSTRAINT check_shipping_method CHECK (shipping_method IS NULL OR shipping_method IN ('OWN_TEAM', 'COURIER')),
    CONSTRAINT orders_cancelled_by_check CHECK (cancelled_by IS NULL OR cancelled_by IN ('CLIENT', 'WAREHOUSE', 'SYSTEM')),
    CONSTRAINT check_cancellation_consistency CHECK (
        ((status = 'CANCELLED') AND (cancellation_reason IS NOT NULL) AND (cancelled_at IS NOT NULL)) OR
        ((status <> 'CANCELLED') AND (cancellation_reason IS NULL) AND (cancelled_at IS NULL))
    )
    -- FOREIGN KEY a user_management.users se omite por la arquitectura modular
);

CREATE SEQUENCE order_management.orders_id_seq START WITH 1 INCREMENT BY 1;


CREATE TABLE order_management.order_items (
    id BIGINT NOT NULL,
    order_id BIGINT NOT NULL,
    product_id BIGINT NOT NULL,
    product_name_snapshot VARCHAR(150) NOT NULL,
    product_image_snapshot VARCHAR(255),
    quantity INT NOT NULL,
    unit_price_snapshot NUMERIC(10,2) NOT NULL,
    -- PostgreSQL GENERATED ALWAYS AS se reemplaza por cálculo en la aplicación o vista
    -- subtotal NUMERIC(10,2) GENERATED ALWAYS AS (quantity * unit_price_snapshot) STORED,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT order_items_pkey PRIMARY KEY (id),
    CONSTRAINT order_items_quantity_check CHECK (quantity > 0),
    CONSTRAINT order_items_unit_price_snapshot_check CHECK (unit_price_snapshot >= 0),
    CONSTRAINT order_items_order_id_fkey FOREIGN KEY (order_id) REFERENCES order_management.orders(id) ON DELETE CASCADE
    -- FOREIGN KEY a product_catalog.products se omite por la arquitectura modular
);

CREATE SEQUENCE order_management.order_items_id_seq START WITH 1 INCREMENT BY 1;


CREATE TABLE order_management.order_status_history (
    id BIGINT NOT NULL,
    order_id BIGINT NOT NULL,
    from_status VARCHAR(50),
    to_status VARCHAR(50) NOT NULL,
    changed_by_user_id BIGINT,
    changed_by_role VARCHAR(20),
    notes TEXT,
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT order_status_history_pkey PRIMARY KEY (id),
    CONSTRAINT order_status_history_changed_by_role_check CHECK (changed_by_role IS NULL OR changed_by_role IN ('CLIENT', 'WAREHOUSE', 'SYSTEM')),
    CONSTRAINT order_status_history_order_id_fkey FOREIGN KEY (order_id) REFERENCES order_management.orders(id) ON DELETE CASCADE
    -- FOREIGN KEY a user_management.users se omite por la arquitectura modular
);

CREATE SEQUENCE order_management.order_status_history_id_seq START WITH 1 INCREMENT BY 1;


-- notification_management
CREATE TABLE notification_management.notification_preferences (
    id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    email_enabled BOOLEAN DEFAULT FALSE NOT NULL,
    whatsapp_enabled BOOLEAN DEFAULT FALSE NOT NULL,
    whatsapp_number VARCHAR(20),
    telegram_enabled BOOLEAN DEFAULT FALSE NOT NULL,
    telegram_chat_id VARCHAR(100),
    sms_enabled BOOLEAN DEFAULT FALSE NOT NULL,
    sms_number VARCHAR(20),
    created_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    updated_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT notification_preferences_pkey PRIMARY KEY (id),
    CONSTRAINT notification_preferences_user_id_key UNIQUE (user_id)
    -- FOREIGN KEY a user_management.users se omite por la arquitectura modular
);

CREATE SEQUENCE notification_management.notification_preferences_id_seq START WITH 1 INCREMENT BY 1;


CREATE TABLE notification_management.notification_logs (
    id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    order_id BIGINT NOT NULL,
    channel VARCHAR(20) NOT NULL,
    status VARCHAR(20) NOT NULL,
    message TEXT,
    error_detail TEXT,
    recipient VARCHAR(255),
    sent_at TIMESTAMP WITHOUT TIME ZONE DEFAULT CURRENT_TIMESTAMP NOT NULL,
    CONSTRAINT notification_logs_pkey PRIMARY KEY (id)
    -- FOREIGN KEY a user_management.users y order_management.orders se omiten por la arquitectura modular
);

CREATE SEQUENCE notification_management.notification_logs_id_seq START WITH 1 INCREMENT BY 1;
